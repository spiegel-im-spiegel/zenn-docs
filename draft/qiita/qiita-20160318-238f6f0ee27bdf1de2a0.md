	"math/rand"

// zundoko-choirs items
const (
	Zun     = "ã‚ºãƒ³"
	Doko    = "ãƒ‰ã‚³"
	Kiyoshi = "ã‚­ãƒ»ãƒ¨ãƒ»ã‚·ï¼"
)

//Choirs - zundoko-choirs list
type Choirs struct {
	c     []string
    dummy int
}

//Push is append choirs
func (c *Choirs) Push(s string) {
	c.c = append(c.c, s) //maybe panic if c is nil.
}

//CountZunDoko returns count of "ZUN" and "DOKO" choirs
func (c *Choirs) CountZunDoko() (int, int) {
	z := 0
	d := 0
	if c == nil {
		return z, d
	}
	for _, s := range c.c {
		switch s {
		case Zun:
			z++
		case Doko:
			d++
		}
	}
	return z, d
}

func (c *Choirs) String() string {
	if c == nil {
		return ""
	} else {
		content := make([]byte, 0, 128)
		for _, s := range c.c {
			content = append(content, s...)
		}
		return string(content)
	}
	return ""
}

func generate() chan string {
	ch := make(chan string)
	go func() {
		zd := [2]string{Zun, Doko}
		rand.Seed(time.Now().UnixNano())
		for {
			ch <- zd[rand.Intn(2)]
		}
	}()
	return ch
}

//Run zundoko-choirs
func Run() *Choirs {
	zd := generate()
	c := &Choirs{c: make([]string, 0)}
	zcount := 0
	for {
		s := <-zd
		c.Push(s)
		if s == Zun {
			zcount++
		} else if zcount >= 4 {
			break
		} else {
			zcount = 0
		}
	}
	c.Push(Kiyoshi)
	return c
}
